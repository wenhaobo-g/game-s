<body>
  <h1>Snake Game</h1>
  <canvas id="snake" width="320" height="320"></canvas>
  <script>
    const ctx=document.getElementById('snake').getContext('2d');
    let snake=[{x:160,y:160}],dx=0,dy=0,fx=80,fy=80;
    setInterval(()=>{
      ctx.fillStyle='black';ctx.fillRect(0,0,320,320);
      snake.unshift({x:snake[0].x+dx,y:snake[0].y+dy});
      if(snake[0].x===fx&&snake[0].y===fy){fx=Math.floor(Math.random()*20)*16;fy=Math.floor(Math.random()*20)*16;}
      else snake.pop();
      ctx.fillStyle='lime';snake.forEach(p=>ctx.fillRect(p.x,p.y,15,15));
      ctx.fillStyle='red';ctx.fillRect(fx,fy,15,15);
    },100);
    document.onkeydown=e=>{
      if(e.key==='ArrowLeft'&&dx===0){dx=-16;dy=0;}
      if(e.key==='ArrowRight'&&dx===0){dx=16;dy=0;}
      if(e.key==='ArrowUp'&&dy===0){dx=0;dy=-16;}
      if(e.key==='ArrowDown'&&dy===0){dx=0;dy=16;}
    }
  </script>
