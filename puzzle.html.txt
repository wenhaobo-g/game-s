<body>
  <h1>Block Puzzle</h1>
  <button class="btn" onclick="startGame()">New Game</button>
  <canvas id="puzzle" width="300" height="300"></canvas>
  <script>
    const canvas=document.getElementById('puzzle');
    const ctx=canvas.getContext('2d');
    let pieces=[];
    function startGame(){
      pieces=[[1,2,3],[4,5,6],[7,8,0]];
      draw();
    }
    function draw(){
      ctx.clearRect(0,0,300,300);
      for(let i=0;i<3;i++){
        for(let j=0;j<3;j++){
          const n=pieces[i][j];
          if(n!==0){
            ctx.fillStyle='#3498db';
            ctx.fillRect(j*100,i*100,98,98);
            ctx.fillStyle='white';
            ctx.font='40px Arial';
            ctx.fillText(n,j*100+35,i*100+60);
          }
        }
      }
    }
    canvas.onclick=e=>{
      const x=Math.floor(e.offsetX/100);
      const y=Math.floor(e.offsetY/100);
      if(pieces[y][x+1]===0){[pieces[y][x],pieces[y][x+1]]=[pieces[y][x+1],pieces[y][x]];}
      if(pieces[y][x-1]===0){[pieces[y][x],pieces[y][x-1]]=[pieces[y][x-1],pieces[y][x]];}
      if(pieces[y+1]?.[x]===0){[pieces[y][x],pieces[y+1][x]]=[pieces[y+1][x],pieces[y][x]];}
      if(pieces[y-1]?.[x]===0){[pieces[y][x],pieces[y-1][x]]=[pieces[y-1][x],pieces[y][x]];}
      draw();
    }
    startGame();
  </script>
